# z1v4
Find dangerous HTTP methods in single click

# Steps:
1. cd ~/tools
2. git clone https://github.com/karthi-the-hacker/z1v4.git
3. cd z1v4
4. chmod +x z1v4
5. chmod +x .banner
6. ./z1v4

# use this for all subdomains
./z1v4

# use this to filter 200 and scanning
./z1v4 live-200

# meg version usage
./z1v4-meg-v
